// ── Config ──────────────────────────────────────────────────────────────
const OWNER = 'beastcooper', REPO = 'MHI_games', BRANCH = 'master', BASE = 'offline';

// Use jsDelivr as the primary URL — it has permissive CORS headers and
// works cross-origin from any page including about:blank.
const JSDELIVR_BASE = `https://cdn.jsdelivr.net/gh/${OWNER}/${REPO}@${BRANCH}/${BASE}/`;

// GitHub API (for directory listing) also works cross-origin.
// Raw GitHub URLs sometimes block non-browser origins; jsDelivr is safer.
const GH_API = p =>
`https://api.github.com/repos/${OWNER}/${REPO}/contents/${p ? BASE+'/'+p : BASE}?ref=${BRANCH}`;

// ── Fetch helper ─────────────────────────────────────────────────────────
// Tries jsDelivr first (great CORS), falls back to raw GitHub.
async function safeFetch(url, fallback) {
const urls = [url, fallback].filter(Boolean);
for (const u of urls) {
try {
const r = await fetch(u, { signal: AbortSignal.timeout(15000) });
if (r.ok) return r;
} catch (e) { console.warn('fetch failed:', u, e.message); }
}
throw new Error('All fetch attempts failed');
}

// ── IndexedDB ────────────────────────────────────────────────────────────
const DB_NAME = 'MHIGamesDB', STORE = 'games';
let db = null;
function initDB() {
return new Promise((res, rej) => {
const req = indexedDB.open(DB_NAME, 1);
req.onerror = () => rej(req.error);
req.onsuccess = () => { db = req.result; res(db); };
req.onupgradeneeded = e => {
if (!e.target.result.objectStoreNames.contains(STORE))
e.target.result.createObjectStore(STORE, { keyPath: 'key' });
};
});
}
const dbGet = key => new Promise((res, rej) => {
const req = db.transaction([STORE],'readonly').objectStore(STORE).get(key);
req.onerror = () => rej(req.error); req.onsuccess = () => res(req.result);
});
const dbPut = (key, content) => new Promise((res, rej) => {
const obj = { key, content, ts: Date.now() };
const req = db.transaction([STORE],'readwrite').objectStore(STORE).put(obj);
req.onerror = () => rej(req.error); req.onsuccess = () => res();
});
const dbClear = () => new Promise((res, rej) => {
const tx = db.transaction([STORE],'readwrite'); tx.objectStore(STORE).clear();
tx.oncomplete = res; tx.onerror = rej;
});
const dbAllKeys = () => new Promise((res, rej) => {
const req = db.transaction([STORE],'readonly').objectStore(STORE).getAllKeys();
req.onerror = () => rej(req.error); req.onsuccess = () => res(req.result);
});

function toggleFav(key, e) {
e.stopPropagation();
favorites = isFav(key) ? favorites.filter(f => f !== key) : [...favorites, key];
saveFavorites(); renderGames();
}
function switchTab(tab) {
currentTab = tab; localStorage.setItem('gamesCurrentTab', tab);
document.getElementById('tab-all').classList.toggle('active', tab === 'all');
document.getElementById('tab-favorites').classList.toggle('active', tab === 'favorites');
document.getElementById('search').style.display = tab === 'all' ? 'block' : 'none';
renderGames();
}

// ── Load game list ───────────────────────────────────────────────────────
async function loadGames() {
try {
const r = await safeFetch(GH_API());
const items = await r.json();
if (!Array.isArray(items)) throw new Error('Unexpected API response');

for (const item of items) {
if (item.type === 'file' && /\.(html|htm)$/i.test(item.name)) {
const key = item.name.replace(/\.(html|htm)$/i, '');
addGame(key, JSDELIVR_BASE + item.name, item.download_url);
}
if (item.type === 'dir') {
try {
const sub = await safeFetch(GH_API(item.name)).then(r => r.json());
const idx = sub.find(f => /^index\.(html|htm)$/i.test(f.name));
if (idx) addGame(item.name, JSDELIVR_BASE + item.name + '/index.html', idx.download_url);
} catch (e) { console.warn('Skipping dir', item.name, e.message); }
}
}
allGames.sort((a, b) => a.key.localeCompare(b.key));
document.getElementById('count').textContent = allGames.length + ' games';
renderGames();
} catch (e) {
console.error('Failed to load games:', e);
document.getElementById('count').textContent = 'Load failed';
document.getElementById('empty-state').style.display = 'block';
document.getElementById('empty-state').textContent = '⚠️ Could not reach GitHub. Check your connection.';
}
}

function addGame(key, url, fallbackUrl) {
if (allGames.some(g => g.key === key)) return;
// Thumbnail: try jsDelivr for the PNG too
const imgUrl = JSDELIVR_BASE + key + '.png';
allGames.push({ key, url, fallbackUrl, imgUrl });
}

// ── Launch a game ────────────────────────────────────────────────────────
async function launchGame(game) {
try {
let html;
const cached = await dbGet(game.key);
if (cached && cached.content) {
html = cached.content;
} else {
const r = await safeFetch(game.url, game.fallbackUrl);
if (!r.ok) throw new Error('HTTP ' + r.status);
html = await r.text();
await dbPut(game.key, html);
cachedSet.add(game.key);
}
const win = window.open('', '_blank');
if (!win) throw new Error('Popup blocked. Please disable popup blocker.');
win.document.open();
win.document.write(html);
win.document.close();
win.document.title = game.key.replace(/[-_]/g, ' ').toUpperCase();
renderGames();
} catch (err) { alert('⚠️ Could not load game\n\n' + err.message); }
}

// ── Planet SVG renderer ──────────────────────────────────────────────────
const RY_SCALE = 0.21, PLANET_R = 68;
const RING_BANDS = [
{ ri: PLANET_R+3, ro: PLANET_R+11, col: 'rgba(70,90,140,0.25)', hov: 'rgba(100,130,180,0.35)' },
{ ri: PLANET_R+12, ro: PLANET_R+24, col: 'rgba(80,110,160,0.35)', hov: 'rgba(110,150,200,0.50)' },
{ ri: PLANET_R+24, ro: PLANET_R+29, col: 'rgba(60,80,120,0.30)', hov: 'rgba(90,120,160,0.45)' },
{ ri: PLANET_R+29, ro: PLANET_R+34, col: 'rgba(8,12,28,0.80)', hov: 'rgba(8,12,28,0.80)' },
{ ri: PLANET_R+34, ro: PLANET_R+46, col: 'rgba(75,105,155,0.40)', hov: 'rgba(110,150,200,0.60)' },
{ ri: PLANET_R+46, ro: PLANET_R+50, col: 'rgba(50,70,110,0.18)', hov: 'rgba(80,110,160,0.28)' },
];
RING_BANDS.forEach(b => { b.riY = b.ri * RY_SCALE; b.roY = b.ro * RY_SCALE; });
const TEXT_BAND = RING_BANDS[4];
const TEXT_RX = (TEXT_BAND.ri + TEXT_BAND.ro) / 2;
const TEXT_RY = TEXT_RX * RY_SCALE;

function annulusPath(cx, cy, ri, riY, ro, roY) {
return [
`M ${cx+ro} ${cy}`,
`A ${ro} ${roY} 0 1 0 ${cx-ro} ${cy}`,
`A ${ro} ${roY} 0 1 0 ${cx+ro} ${cy}`,
`M ${cx+ri} ${cy}`,
`A ${ri} ${riY} 0 1 1 ${cx-ri} ${cy}`,
`A ${ri} ${riY} 0 1 1 ${cx+ri} ${cy}`,
'Z'
].join(' ');
}

function makePlanetSVG(key, imgUrl) {
const ns = 'http://www.w3.org/2000/svg';
const id = 'p' + Math.random().toString(36).slice(2, 9);
const W = 280, H = 190, cx = W/2, cy = H/2;

const svg = document.createElementNS(ns, 'svg');
svg.setAttribute('viewBox', `0 0 ${W} ${H}`);
svg.setAttribute('width', W); svg.setAttribute('height', H);
svg.classList.add('planet-svg'); svg.style.cursor = 'pointer';

const defs = document.createElementNS(ns, 'defs');

const cp = document.createElementNS(ns, 'clipPath');
cp.setAttribute('id', id+'_cp');
const cpc = document.createElementNS(ns, 'circle');
cpc.setAttribute('cx', cx); cpc.setAttribute('cy', cy); cpc.setAttribute('r', PLANET_R);
cp.appendChild(cpc); defs.appendChild(cp);

const bkClip = document.createElementNS(ns, 'clipPath');
bkClip.setAttribute('id', id+'_bk');
const bkR = document.createElementNS(ns, 'rect');
bkR.setAttribute('x', 0); bkR.setAttribute('y', cy);
bkR.setAttribute('width', W); bkR.setAttribute('height', cy+2);
bkClip.appendChild(bkR); defs.appendChild(bkClip);

const frClip = document.createElementNS(ns, 'clipPath');
frClip.setAttribute('id', id+'_fr');
const frR = document.createElementNS(ns, 'rect');
frR.setAttribute('x', 0); frR.setAttribute('y', 0);
frR.setAttribute('width', W); frR.setAttribute('height', cy+1);
frClip.appendChild(frR); defs.appendChild(frClip);

const specGrad = document.createElementNS(ns, 'radialGradient');
specGrad.setAttribute('id', id+'_sp');
specGrad.setAttribute('cx', '36%'); specGrad.setAttribute('cy', '32%'); specGrad.setAttribute('r', '62%');
[['0%','rgba(255,255,255,0.20)'],['45%','rgba(255,255,255,0.03)'],['100%','rgba(0,0,0,0.50)']].forEach(([o,c]) => {
const s = document.createElementNS(ns, 'stop');
s.setAttribute('offset', o); s.setAttribute('stop-color', c); specGrad.appendChild(s);
});
defs.appendChild(specGrad);

const k = 0.5523;
const arcD =
`M ${cx-TEXT_RX},${cy} ` +
`C ${cx-TEXT_RX},${cy-TEXT_RY*k} ${cx-TEXT_RX*k},${cy-TEXT_RY} ${cx},${cy-TEXT_RY} ` +
`C ${cx+TEXT_RX*k},${cy-TEXT_RY} ${cx+TEXT_RX},${cy-TEXT_RY*k} ${cx+TEXT_RX},${cy}`;
const arcPath = document.createElementNS(ns, 'path');
arcPath.setAttribute('id', id+'_arc'); arcPath.setAttribute('d', arcD); arcPath.setAttribute('fill', 'none');
defs.appendChild(arcPath);
svg.appendChild(defs);

// rings behind
const backG = document.createElementNS(ns, 'g');
backG.setAttribute('clip-path', `url(#${id}_bk)`);
backG.classList.add('rings-back');
RING_BANDS.forEach((b, i) => {
const path = document.createElementNS(ns, 'path');
path.setAttribute('d', annulusPath(cx, cy, b.ri, b.riY, b.ro, b.roY));
path.setAttribute('fill', b.col); path.setAttribute('fill-rule', 'evenodd');
path.setAttribute('opacity', '0.45'); path.dataset.bi = i;
backG.appendChild(path);
});
svg.appendChild(backG);

// planet base
const base = document.createElementNS(ns, 'circle');
base.setAttribute('cx', cx); base.setAttribute('cy', cy); base.setAttribute('r', PLANET_R);
base.setAttribute('fill', '#00008B'); svg.appendChild(base);

// thumbnail image (graceful fallback to blue circle)
const img = document.createElementNS(ns, 'image');
img.setAttribute('href', imgUrl);
img.setAttribute('x', cx-PLANET_R); img.setAttribute('y', cy-PLANET_R);
img.setAttribute('width', PLANET_R*2); img.setAttribute('height', PLANET_R*2);
img.setAttribute('clip-path', `url(#${id}_cp)`);
img.setAttribute('preserveAspectRatio', 'xMidYMid slice');
img.addEventListener('error', () => { img.style.display = 'none'; });
svg.appendChild(img);

// specular
const spec = document.createElementNS(ns, 'circle');
spec.setAttribute('cx', cx); spec.setAttribute('cy', cy); spec.setAttribute('r', PLANET_R);
spec.setAttribute('fill', `url(#${id}_sp)`); spec.setAttribute('clip-path', `url(#${id}_cp)`);
svg.appendChild(spec);

// rim
const rim = document.createElementNS(ns, 'circle');
rim.setAttribute('cx', cx); rim.setAttribute('cy', cy); rim.setAttribute('r', PLANET_R);
rim.setAttribute('fill', 'none'); rim.setAttribute('stroke', 'rgba(140,200,255,0.20)');
rim.setAttribute('stroke-width', '1'); svg.appendChild(rim);

// rings in front
const frontG = document.createElementNS(ns, 'g');
frontG.setAttribute('clip-path', `url(#${id}_fr)`);
frontG.classList.add('rings-front');
RING_BANDS.forEach((b, i) => {
const path = document.createElementNS(ns, 'path');
path.setAttribute('d', annulusPath(cx, cy, b.ri, b.riY, b.ro, b.roY));
path.setAttribute('fill', b.col); path.setAttribute('fill-rule', 'evenodd');
path.dataset.bi = i; frontG.appendChild(path);
});
svg.appendChild(frontG);

// ring label
const label = key.replace(/[-_]/g, ' ').toUpperCase();
const textEl = document.createElementNS(ns, 'text');
textEl.setAttribute('font-family', "'Courier New', monospace");
textEl.setAttribute('font-size', '10');
textEl.setAttribute('font-weight', '700');
textEl.setAttribute('letter-spacing', '2');
textEl.setAttribute('clip-path', `url(#${id}_fr)`);
textEl.classList.add('ring-label');
const tp = document.createElementNS(ns, 'textPath');
tp.setAttributeNS('http://www.w3.org/1999/xlink', 'href', '#'+id+'_arc');
tp.setAttribute('href', '#'+id+'_arc');
tp.setAttribute('startOffset', '50%');
tp.setAttribute('text-anchor', 'middle');
tp.setAttribute('fill', 'rgba(200,235,255,0.85)');
tp.classList.add('ring-label-tp');
tp.textContent = label;
textEl.appendChild(tp); svg.appendChild(textEl);
svg.dataset.id = id;
return svg;
}

// ── Render grid ──────────────────────────────────────────────────────────
function renderGames() {
const q = document.getElementById('search').value.toLowerCase();
let list = allGames;
if (currentTab === 'favorites') list = allGames.filter(g => isFav(g.key));
else if (q) list = allGames.filter(g => g.key.replace(/[-_]/g,' ').toLowerCase().includes(q));

const grid = document.getElementById('games');
const empty = document.getElementById('empty-state');
grid.innerHTML = '';
if (!list.length) { empty.style.display = 'block'; return; }
empty.style.display = 'none';

for (const game of list) {
const wrap = document.createElement('div');
wrap.className = 'game-wrap';

const svgEl = makePlanetSVG(game.key, game.imgUrl);
svgEl.addEventListener('click', () => launchGame(game));
wrap.appendChild(svgEl);

wrap.addEventListener('mouseenter', () => {
svgEl.querySelectorAll('.rings-front path').forEach((p, i) => {
if (i < RING_BANDS.length) p.setAttribute('fill', RING_BANDS[i].hov);
});
svgEl.querySelectorAll('.ring-label-tp').forEach(tp => tp.setAttribute('fill', 'rgba(5,10,30,1)'));
});
wrap.addEventListener('mouseleave', () => {
svgEl.querySelectorAll('.rings-front path').forEach((p, i) => {
if (i < RING_BANDS.length) p.setAttribute('fill', RING_BANDS[i].col);
});
svgEl.querySelectorAll('.ring-label-tp').forEach(tp => tp.setAttribute('fill', 'rgba(200,235,255,0.85)'));
});

const star = document.createElement('button');
star.className = 'star-btn' + (isFav(game.key) ? ' favorited' : '');
star.textContent = isFav(game.key) ? '★' : '☆';
star.onclick = e => toggleFav(game.key, e);
wrap.appendChild(star);

const download = document.createElement('button');
download.className = 'download-btn' + (cachedSet.has(game.key) ? ' cached' : '');
download.textContent = cachedSet.has(game.key) ? '✓' : '⬇';
download.onclick = e => downloadGame(game, download, e);
wrap.appendChild(download);

if (cachedSet.has(game.key)) {
const badge = document.createElement('div');
badge.className = 'cached-badge visible'; badge.textContent = '✓';
wrap.appendChild(badge);
}
grid.appendChild(wrap);
}
}

function filterGames() { renderGames(); }

// ── Download / cache ─────────────────────────────────────────────────────
async function downloadGame(game, btn, e) {
e.stopPropagation();
if (cachedSet.has(game.key)) return;
btn.disabled = true; btn.textContent = '⏳';
try {
const r = await safeFetch(game.url, game.fallbackUrl);
if (!r.ok) throw new Error('HTTP ' + r.status);
const html = await r.text();
await dbPut(game.key, html);
cachedSet.add(game.key);
btn.textContent = '✓'; btn.classList.add('cached');
await updateStorage(); renderGames();
} catch (err) {
console.error('Failed to download', game.key, err);
btn.textContent = '⬇'; btn.disabled = false;
}
}

async function downloadAllGames() {
const btn = document.getElementById('downloadAllBtn');
const fill = document.getElementById('progress-fill');
btn.disabled = true; btn.textContent = '⬇️ Downloading…';
let done = 0;
for (let i = 0; i < allGames.length; i++) {
const g = allGames[i];
try {
if (!cachedSet.has(g.key)) {
const r = await safeFetch(g.url, g.fallbackUrl);
if (r.ok) { await dbPut(g.key, await r.text()); cachedSet.add(g.key); }
}
done++;
} catch (e) { console.warn('Skip', g.key); }
fill.style.width = ((i + 1) / allGames.length * 100) + '%';
}
btn.disabled = false; btn.textContent = `✅ ${done}/${allGames.length} Cached`;
setTimeout(() => { btn.textContent = '⬇️ Download All'; fill.style.width = '0%'; }, 4000);
await updateStorage(); renderGames();
}

async function clearCache() {
if (!confirm('Delete all cached games?')) return;
await dbClear(); cachedSet.clear();
alert('✅ Cache cleared!'); await updateStorage(); renderGames();
}

async function updateStorage() {
const el = document.getElementById('storage-info');
if (navigator.storage && navigator.storage.estimate) {
const e = await navigator.storage.estimate();
el.textContent = `Storage: ${(e.usage/1024/1024).toFixed(1)} MB / ${(e.quota/1024/1024).toFixed(0)} MB • ${cachedSet.size} games cached`;
} else { el.textContent = `${cachedSet.size} games cached`; }
}

// ── Splash animation ─────────────────────────────────────────────────────
function initSplash() {
const symbols = ['∑','∫','π','∞','√','Δ','∂','λ','θ','Ω','±','≈','≠','∝','∈'];
const sc = document.getElementById('mathSymbols');
for (let i = 0; i < 15; i++) {
const s = document.createElement('span');
s.className = 'math-symbol';
s.textContent = symbols[i % symbols.length];
s.style.cssText = `left:${5+Math.random()*90}%;top:${60+Math.random()*40}%;animation-delay:${Math.random()*8}s;animation-duration:${6+Math.random()*6}s;font-size:${1+Math.random()*1.5}rem`;
sc.appendChild(s);
}
const pc = document.getElementById('particles');
['#00d4ff','#7b2fff','#e0e8ff','#00ffaa'].forEach((col, ci) => {
for (let i = ci; i < 30; i += 4) {
const p = document.createElement('div'); p.className = 'particle';
p.style.background = col; p.style.left = '50%'; p.style.top = '40%';
const angle = (Math.PI*2*i)/30, dist = 80+Math.random()*200;
const tx = Math.cos(angle)*dist, ty = Math.sin(angle)*dist, dur = 1+Math.random()*0.5;
p.animate([
{ opacity:0, transform:'translate(-50%,-50%) scale(0)' },
{ opacity:1, transform:`translate(calc(-50% + ${tx}px),calc(-50% + ${ty}px)) scale(1)`, offset:0.3 },
{ opacity:0, transform:`translate(calc(-50% + ${tx*1.5}px),calc(-50% + ${ty*1.5}px)) scale(0)` }
], { duration:dur*1000, delay:800, easing:'cubic-bezier(0.16,1,0.3,1)', fill:'forwards' });
pc.appendChild(p);
}
});
setTimeout(() => {
document.getElementById('splash-screen').classList.add('hidden');
document.getElementById('games-screen').classList.add('visible');
}, 3000);
}
